241  if (current_pass_type == PassType::Declare) {
242  NLANG_ASSERT(!node.meta);
243  scope_stack.push_front(IntrusivePtr<Scope>(new Scope(scope_stack.empty() ? nullptr : scope_stack.front().get(), weak)));
