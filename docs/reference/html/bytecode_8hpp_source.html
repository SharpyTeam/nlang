261  template<typename Operand>
262  static std::function<UString(Instruction)> GetPrinter() {
263  if constexpr (std::is_same_v<Operand, Register>) {
264  return [](Instruction instruction) -> UString { return std::to_string(instruction.reg); };
265  } else if constexpr (std::is_same_v<Operand, ContextDescriptor>) {
266  return [](Instruction instruction) -> UString { return std::to_string(instruction.context_descriptor.index) + " " + std::to_string(instruction.context_descriptor.depth); };
267  } else if constexpr (std::is_same_v<Operand, RegistersRange>) {
268  return [](Instruction instruction) -> UString { return std::to_string(instruction.reg_range.first) + " " + std::to_string(instruction.reg_range.count); };
269  } else if constexpr (std::is_same_v<Operand, Offset>) {
270  return [](Instruction instruction) -> UString { return std::to_string(instruction.offset); };
271  } else if constexpr (std::is_same_v<Operand, ConstantIndex>) {
272  return [](Instruction instruction) -> UString { return std::to_string(instruction.const_index); };
273  } else if constexpr (std::is_same_v<Operand, ImmediateDouble>) {
274  return [](Instruction instruction) -> UString { return std::to_string(instruction.immediate_double); };
275  } else if constexpr (std::is_same_v<Operand, ImmediateInt32>) {
276  return [](Instruction instruction) -> UString { return std::to_string(instruction.immediate_int32); };
